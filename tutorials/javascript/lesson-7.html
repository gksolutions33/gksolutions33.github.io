Creating Functions

function greet() {
  console.log("Hello!");
}

greet();   // Call the function

Functions with Parameters

function greet(name) {
  console.log("Hello, " + name);
}

greet("Alice");
greet("Bob");

Return Values

function add(a, b) {
  return a + b;
}

let result = add(5, 3);
console.log(result);  // 8

Arrow Functions

const multiply = (a, b) => a * b;
console.log(multiply(4, 5));  // 20
