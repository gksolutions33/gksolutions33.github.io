Media Queries

/* Mobile first */
p { font-size: 14px; }

/* Tablets and up */
@media (min-width: 768px) {
  p { font-size: 16px; }
}

/* Desktop and up */
@media (min-width: 1024px) {
  p { font-size: 18px; }
}
